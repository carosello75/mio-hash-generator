Esempi di Utilizzo

JavaScript/Fetch
// Genera hash per una password
     const response = await fetch('/api/generate-hash', {
         method: 'POST',
         headers: {
             'Content-Type': 'application/json',
         },
         body: JSON.stringify({
             input_text: 'MySecurePassword123',
             salt: 'random_salt_here',
             iterations: 10,
             use_timestamp: false
         })
     });
     
     const data = await response.json();
     console.log('SHA-256 Hash:', data.hashes.sha256.hash);
Python/Requests
import requests
     
     # Genera salt
     salt_response = requests.get('https://tuodominio.com/api/generate-salt')
     salt = salt_response.json()['salt']
     
     # Genera hash
     hash_response = requests.post('https://tuodominio.com/api/generate-hash', json={
         'input_text': 'MySecurePassword123',
         'salt': salt,
         'iterations': 10,
         'use_timestamp': False
     })
     
     hashes = hash_response.json()['hashes']
     print(f"SHA-256: {hashes['sha256']['hash']}")
cURL
